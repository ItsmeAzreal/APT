# === Core PyTorch and Ecosystem ===
torch               # Core deep learning framework (CPU+GPU)
torchvision         # Image utilities/models (optional, vision tasks)
torchaudio          # Audio utilities/models (optional)

# === Training & Experiment Management ===
tensorboard         # Training visualization/logging dashboard
tqdm                # Fast, pretty progress bars

flash-attn          # Ultra-fast transformer attention kernels (optional for LLMs, only works on some GPUs)
bitsandbytes        # 8-bit (and 4-bit) optimizers/quantization (optional, for speed/memory savings)

transformers        # Hugging Face models, pipelines, tokenizers
datasets            # Hugging Face datasets (streaming/loading)
accelerate          # Easy multi-GPU/TPU setup (useful for DDP or FSDP)
safetensors         # Fast, secure model checkpoints

huggingface-hub     # Upload/download models/datasets from Hugging Face Hub

numpy               # General math and array library
matplotlib          # Visualization/plotting
